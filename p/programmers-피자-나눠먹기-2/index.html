function gcd(a, b) {
  let result;

  while (b != 0) {
    result = a % b;
    a = b;
    b = result;
  }
  return a;
}

function solution(n) {
  let defaultSlice = 6;
  let person = n;
  let lcm;

  if (person < defaultSlice) {
    lcm = (person * defaultSlice) / gcd(defaultSlice, person);
    return lcm / 6;
  } else {
    lcm = (person * defaultSlice) / gcd(person, defaultSlice);
    return lcm / 6;
  }
}
